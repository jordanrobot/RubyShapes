  def gauge_converter
    if @t > 1 then
      case @t
      when 20
        @t = 0.035
      when 18
        @t = 0.049
      when 16
        @t = 0.065
      when 14
        @t = 0.083
      when 13
        @t = 0.095
      when 12
        @t = 0.109
      when 11
        @t = 0.120
      when 10
        @t = 0.134
      when 9
        @t = 0.148
      when 8
        @t = 0.165
      when 7
        @t = 0.180
      when 6
        @t = 0.203
      when 5
        @t = 0.220
      when 4
        @t = 0.238
      when 3
        @t = 0.259
      when 2
        @t = 0.284
      when 1
        @t = 0.300
      end
    if @t > 20
      puts "Please enter an accurate thickness, either as a decimal number or as a gauge number"
      exit
    end
    @t = @t.to_s.to_d
    end
  end
